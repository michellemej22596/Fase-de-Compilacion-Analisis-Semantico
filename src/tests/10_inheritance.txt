class Forma {
    function dibujar() {
        print("Dibujando forma");
    }
}

class Circulo : Forma {
    function dibujar() {
        print("Dibujando círculo");
    }
}

class Rectangulo : Forma {
    function dibujar() {
        print("Dibujando rectángulo");
    }
}

let c: Circulo = new Circulo();
let r: Rectangulo = new Rectangulo();

c.dibujar();
r.dibujar();
